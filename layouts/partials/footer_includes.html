{{/* Add all JS files from the components directory to the bundle */}} {{- range (readDir "/assets/js/components") }} {{- $rsrc := resources.Get (printf "js/components/%v" .Name) -}} {{- $.Scratch.Add "jsFiles" (slice $rsrc) -}} {{- end }} {{- .Scratch.Add "jsFiles" (slice (resources.Get "js/main.js")) -}} {{- $allJS := (.Scratch.Get "jsFiles") | resources.Concat "js/main.js" -}} {{- $bundleJS := $allJS | resources.Minify | resources.Fingerprint }}
